function createHeart(e) { let heart = document.createElement("div"); heart.innerHTML = "❤️"; heart.classList.add("heart"); heart.style.left = `${e.pageX - 15}px`; // Adjust to mouse position heart.style.top = `${e.pageY - 15}px`; // Adjust to mouse position heart.style.animationDuration = (Math.random() * 3 + 2) + "s"; document.body.appendChild(heart); // Create message let message = document.createElement("div"); message.classList.add("message"); message.innerText = "I love you"; message.style.left = `${e.pageX - 60}px`; message.style.top = `${e.pageY - 50}px`; document.body.appendChild(message); // Make the message fade in and out setTimeout(() => { message.style.opacity = 1; // Fade in }, 100); setTimeout(() => { message.style.opacity = 0; // Fade out // Remove message after fade out setTimeout(() => message.remove(), 1500); }, 1500); // Remove heart after animation setTimeout(() => heart.remove(), 2000); }
